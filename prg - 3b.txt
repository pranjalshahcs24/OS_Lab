#include <stdio.h>

#define MAX 20
#define INF 99999

int main() {
    int n;
    int arrival[MAX], burst[MAX], deadline[MAX];
    int remaining[MAX];
    int completed = 0, time = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    for (int i = 0; i < n; i++) {
        printf("Process %d:\n", i + 1);
        printf("Arrival Time: ");
        scanf("%d", &arrival[i]);
        printf("Burst Time: ");
        scanf("%d", &burst[i]);
        printf("Deadline: ");
        scanf("%d", &deadline[i]);

        remaining[i] = burst[i];
    }

    printf("\nEDF Scheduling:\n");

    while (completed < n) {
        int min_deadline = INF;
        int selected = -1;

        // Find process with earliest deadline
        for (int i = 0; i < n; i++) {
            if (arrival[i] <= time && remaining[i] > 0) {
                if (deadline[i] < min_deadline) {
                    min_deadline = deadline[i];
                    selected = i;
                }
            }
        }

        if (selected == -1) {
            printf("%d ms: CPU is idle\n", time);
            time++;
        } else {
            printf("%d ms: Task %d is running\n", time, selected + 1);

            remaining[selected]--;
            time++;

            if (remaining[selected] == 0) {
                completed++;
            }
        }
    }

    return 0;
}
