#include <stdio.h>

#define MAX 10

int gcd(int a, int b) {
    if (b == 0)
        return a;
    return gcd(b, a % b);
}

int lcm(int a, int b) {
    return (a * b) / gcd(a, b);
}

int find_lcm(int arr[], int n) {
    int result = arr[0];
    for (int i = 1; i < n; i++) {
        result = lcm(result, arr[i]);
    }
    return result;
}

int main() {
    int n;
    int C[MAX], P[MAX], remaining[MAX];
    int hyperperiod;

    printf("Enter number of tasks: ");
    scanf("%d", &n);

    printf("Enter computation times:\n");
    for (int i = 0; i < n; i++) {
        scanf("%d", &C[i]);
        remaining[i] = 0;
    }

    printf("Enter periods:\n");
    for (int i = 0; i < n; i++) {
        scanf("%d", &P[i]);
    }

    
    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            if (P[i] > P[j]) {
                int temp;

                temp = P[i]; P[i] = P[j]; P[j] = temp;
                temp = C[i]; C[i] = C[j]; C[j] = temp;
            }
        }
    }

    hyperperiod = find_lcm(P, n);
    printf("\nHyperperiod = %d\n", hyperperiod);

    printf("\nTime\tTask\n");

    for (int time = 0; time < hyperperiod; time++) {

       
        for (int i = 0; i < n; i++) {
            if (time % P[i] == 0) {
                remaining[i] = C[i];
            }
        }

        
        int selected = -1;
        for (int i = 0; i < n; i++) {
            if (remaining[i] > 0) {
                selected = i;
                break;
            }
        }

        if (selected != -1) {
            printf("%d\tT%d\n", time, selected + 1);
            remaining[selected]--;
        } else {
            printf("%d\tIdle\n", time);
        }
    }

    return 0;
}
